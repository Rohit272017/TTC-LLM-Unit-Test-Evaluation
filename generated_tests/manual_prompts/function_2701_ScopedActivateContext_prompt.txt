  auto* tls = &tls_data;
  if (tls->depth == 0) {
    VLOG(3) << "ScopedActivateContext switching to "
            << gpu_context->device_ordinal();
    gpu_context->SetActive();
    tls->depth = 1;
    tls->device_ordinal = gpu_context->device_ordinal();

    to_restore_ = nullptr;
    return;
  }
  tls->depth++;

  to_restore_ = tls->context;
  gpu_context->SetActive();
  tls->device_ordinal = gpu_context->device_ordinal();
  tls->context = gpu_context;
}
